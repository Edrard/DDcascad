Base Usage


$('.test').ddcascad({
    object       : tank_list,
    objectShow   : 'en',
});

Predefine Selected


$('.test2').ddcascad({
    current : ['usa','heavyTank','14881'],
    object       : tank_list,
    objectShow   : 'en',
});

Custome Function on last dropdown select


$('.test3').ddcascad({
    object       : tank_list,
    objectShow   : 'en',
    onLastChoise : function(state, settings, e){
        alert($(this).find("option:selected").text());
    }
});

Using delimiter


$('.test4').ddcascad({
    object       : tank_list,
    objectShow   : 'en',
    delimiter    : '<br>';
});

Additional attributes in options

Adding data-lvl attribute if lvl key exist


$('.test5').ddcascad({
    object       : tank_list,
    objectShow   : 'en',
    moreData     : { 'lvl' : 'data-lvl'},
});
